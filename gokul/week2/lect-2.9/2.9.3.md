time = input("Enter time (longer/shorter): ").lower()
price = input("Enter price (lower/higher): ").lower()

print("Travel from City A to City B")

if time == "longer":
    if price == "lower":
        print("Travel by Coach")
    elif price == "higher":
        print("Travel by Train")
    else:
        print("Invalid price input")

elif time == "shorter":
    if price == "lower":
        print("Travel by Red Eye Flight")
    elif price == "higher":
        print("Travel by Daytime Flight")
    else:
        print("Invalid price input")

else:
    print("Invalid time input")

print("Arrived at City B")
