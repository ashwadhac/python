num = int(input("Enter a number: "))

print("Prime numbers less than", num, "are:")

for n in range(2, num):  # check all numbers from 2 to num-1
    is_prime = True
    for i in range(2, n):  # check if n is divisible by any number from 2 to n-1
        if n % i == 0:
            is_prime = False
            break
    if is_prime:
        print(n, end=' ')
